  // Process stringRaw to ensure it contains exactly 5 comma-delimited fields

  while (stringRaw.split(',').length > 5) {
    // Find the index of the 3rd comma
    let commaCount = 0;
    let thirdCommaIndex = -1;

    for (let i = 0; i < stringRaw.length; i++) {
      if (stringRaw[i] === ',') {
        commaCount++;
        if (commaCount === 3) {
          thirdCommaIndex = i;
          break;
        }
      }
    }

    if (thirdCommaIndex !== -1) {
      // Replace the 3rd comma with a hyphen

      break; // Safety check: if we can't find a 3rd comma, exit the loop
    }
  }

  }

  const result = parse(stringRaw, {
    raw: true,
    trim: true,
    onRecord: onRecord,

  return result[0];
}

export function fixInconsistentColumns(raw: string): string {
  // find the 3rd comma in the line and replace it with a dash it as it's probably incorrectly
  // part of the payee's name (and shouldn't be but HSBC are crap so...)

  // Process raw to ensure it contains exactly 5 comma-delimited fields
  // by repeatedly replacing the 3rd comma with a hyphen until we have 5 fields
  while (raw.split(',').length > 5) {
    // Find the index of the 3rd comma
    let commaCount = 0;
    let thirdCommaIndex = -1;

    for (let i = 0; i < raw.length; i++) {
      if (raw[i] === ',') {
        commaCount++;
        if (commaCount === 3) {
          thirdCommaIndex = i;
          break;
        }
      }
    }

    if (thirdCommaIndex !== -1) {
      // Replace the 3rd comma with a hyphen
      raw = raw.slice(0, thirdCommaIndex) + '-' + raw.slice(thirdCommaIndex + 1);
    } else {
      break; // Safety check: if we can't find a 3rd comma, exit the loop
    }
  }

  // If we couldn't process it correctly, return an empty string
  if (raw.split(',').length !== 5) {
    return '';
  }

  return raw;
}
